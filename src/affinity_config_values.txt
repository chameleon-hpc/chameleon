compile with extra options example:
	export CHAM_AFF_TASK_SELECTION_STRAT=1
	export CHAM_AFF_PAGE_SELECTION_STRAT=5
	export CHAM_AFF_PAGE_WEIGHTING_STRAT=2
	CUSTOM_COMPILE_FLAGS="-DUSE_TASK_AFFINITY=1" make

======================
flags
======================
USE_TASK_AFFINITY=1 to use task affinity (default 1)

======================
environment variables:
======================
CHAM_AFF_TASK_SELECTION_STRAT= (default 0)
	//just pop the first task
  CHAM_AFF_NONE = 0,
  //start at the front and search each task
  //pop the first with the correct domain
  //else pop the first task
  CHAM_AFF_ALL_LINEAR = 1

CHAM_AFF_PAGE_SELECTION_STRAT= (default 0)
	cham_affinity_page_mode_first_page_of_first_affinity_only = 0,
  cham_affinity_page_mode_divide_in_n_pages = 1,
  cham_affinity_page_mode_every_nth_page = 2,
  cham_affinity_page_mode_first_and_last_page = 3,
  cham_affinity_page_mode_continuous_binary_search = 4,
  cham_affinity_page_mode_first_page = 5,

CHAM_AFF_PAGE_WEIGHTING_STRAT= (default 0)
	cham_affinity_page_weight_mode_first_page_only = 0,
  cham_affinity_page_weight_mode_majority = 1,
  cham_affinity_page_weight_mode_by_size = 2,

CHAM_AFF_CONSIDER_TYPES= (default 0) //which types should be considered when calculating the domain of a task?
  CONSIDER_ALL_TYPES = 0, //considers all except literals
  CONSIDER_TYPE_TO = 1 //considers only the type to
