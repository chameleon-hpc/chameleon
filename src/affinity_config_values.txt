compile with extra options example:

CUSTOM_COMPILE_FLAGS="-DUSE_TASK_AFFINITY=1 -DTASK_AFFINITY_DEBUG=1" make

export CHAM_AFF_TASK_SELECTION_STRAT=3
export CHAM_AFF_PAGE_SELECTION_STRAT=8
export CHAM_AFF_PAGE_WEIGHTING_STRAT=2
export CHAM_AFF_CONSIDER_TYPES=1
export CHAM_AFF_PAGE_SELECTION_N=3
export CHAM_AFF_TASK_SELECTION_N=8
export CHAM_AFF_MAP_MODE=0
export CHAM_AFF_ALWAYS_CHECK_PHYSICAL=0

======================
flags
======================
USE_TASK_AFFINITY=1 to use task affinity (default 1)

TASK_AFFINITY_DEBUG=1 to activate debug prints (default 0)

comile with affinity debug: make aff_debug

======================
environment variables:
======================
CHAM_AFF_TASK_SELECTION_STRAT= (default 0)
	//just pop the first task
  CHAM_AFF_NONE = 0,
  //start at the front and search each task
  //pop the first with the correct domain
  //else pop the first task
  CHAM_AFF_ALL_LINEAR = 1
  CHAM_AFF_FIRST_N = 2,
  // check N tasks euqally spaced
  CHAM_AFF_N_EQS = 3,
  // check tasks linearely with stride N
  CHAM_AFF_EVERY_NTH = 4

CHAM_AFF_PAGE_SELECTION_STRAT= (default 0)
	cham_affinity_page_mode_first_page_of_first_affinity_only = 0,
  cham_affinity_page_mode_divide_in_n_pages = 1,
  cham_affinity_page_mode_every_nth_page = 2,
  cham_affinity_page_mode_first_and_last_page = 3,
  cham_affinity_page_mode_continuous_binary_search = 4,
  cham_affinity_page_mode_first_page = 5,
  cham_affinity_page_mode_first_page_of_n_affinities_eqs = 6
  cham_affinity_page_mode_middle_page_of_n_affinities_eqs = 7
  cham_affinity_page_mode_middle_page = 8

CHAM_AFF_PAGE_WEIGHTING_STRAT= (default 0)
	cham_affinity_page_weight_mode_first_page_only = 0,
  cham_affinity_page_weight_mode_majority = 1,
  cham_affinity_page_weight_mode_by_size = 2,
  cham_affinity_page_weight_mode_by_size_old = 3

CHAM_AFF_CONSIDER_TYPES= (default 0) //which types should be considered when calculating the domain of a task?
  CONSIDER_ALL_TYPES = 0, //considers all except literals
  CONSIDER_TYPE_TO = 1 //considers only the type to

CHAM_AFF_PAGE_SELECTION_N= (default 3) //parameter to use for page selection strategy, e.g. number of pages to check

CHAM_AFF_TASK_SELECTION_N= (default 3) //parameter for task selection strategies

CHAM_AFF_MAP_MODE= (default 0)
  CHAM_AFF_DOMAIN_MODE = 0,
  CHAM_AFF_TEMPORAL_MODE = 1

CHAM_AFF_ALWAYS_CHECK_PHYSICAL= (default 0)
  off (never update domain) = 0
  on (always update domain) = 1
